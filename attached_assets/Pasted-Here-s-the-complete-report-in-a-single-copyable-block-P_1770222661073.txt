Fix 2: Expense Breakdown
javascript// Don't add phantom categories
function calculateBurnBreakdown(userInput) {
  const breakdown = {
    payroll: userInput.payroll || 0,
    marketing: userInput.marketing || 0,
    operating: userInput.operating || 0,
    cogs: userInput.cogs || 0,
    other: userInput.other || 0
  };
  
  breakdown.total = Object.values(breakdown).reduce((a, b) => a + b, 0);
  
  // Calculate percentages
  Object.keys(breakdown).forEach(key => {
    if (key !== 'total') {
      breakdown[`${key}Pct`] = (breakdown[key] / breakdown.total * 100).toFixed(1);
    }
  });
  
  return breakdown;
}
Fix 3: Revenue Growth
javascript// Use user input when no historical data
function calculateRevenueGrowth(userInput, revenueHistory) {
  if (revenueHistory && revenueHistory.length >= 2) {
    // Calculate from actual data
    const current = revenueHistory[revenueHistory.length - 1];
    const previous = revenueHistory[revenueHistory.length - 2];
    return ((current - previous) / previous * 100).toFixed(1);
  }
  
  // Fall back to user input
  return userInput.monthlyGrowthRate || 0;
}
