DRY_RUN = "--dry-run" in sys.argv

RESULTS = {"applied": 0, "skipped": 0, "failed": 0, "errors": []}


# ── Utilities ──────────────────────────────────────────────────────────────

def backup(filepath):
    """Create a backup of the file before patching."""
    if DRY_RUN:
        return
    os.makedirs(BACKUP_DIR, exist_ok=True)
    backup_path = os.path.join(BACKUP_DIR, filepath.replace("/", "__").replace("\\", "__"))
    if not os.path.exists(backup_path):  # Don't overwrite existing backup
        shutil.copy2(filepath, backup_path)


def restore_all():
    """Restore all files from backups."""
    if not os.path.exists(BACKUP_DIR):
        print("No backups found.")
        return
    for backup_file in os.listdir(BACKUP_DIR):
        original = backup_file.replace("__", "/")
        backup_path = os.path.join(BACKUP_DIR, backup_file)
        if os.path.exists(backup_path):
            shutil.copy2(backup_path, original)
            print(f"  RESTORED: {original}")
    print("All files restored from backups.")


def read_file(filepath):
    """Read file content, return None if missing."""
    try:
        with open(filepath, "r", encoding="utf-8") as f:
            return f.read()
    except FileNotFoundError:
        RESULTS["errors"].append(f"File not found: {filepath}")
        return None
    except Exception as e:
        RESULTS["errors"].append(f"Read error {filepath}: {e}")
        return None


def write_file(filepath, content):
    """Write file content."""
    if DRY_RUN:
        return True
    try:
        with open(filepath, "w", encoding="utf-8") as f:
            f.write(content)
        return True
    except Exception as e:
        RESULTS["errors"].append(f"Write error {filepath}: {e}")
        return False


def patch(filepath, old, new, description, count=1):
    """
    Apply a string replacement patch.
    count=0 means replace ALL occurrences.
    Returns True if applied, False if skipped/failed.
    """
    content = read_file(filepath)
    if content is None:
        RESULTS["failed"] += 1
        return False

    if old not in content:
        # Check if already patched
        if new in content:
            print(f"  SKIP (already applied): {description}")
            RESULTS["skipped"] += 1
            return True
        print(f"  SKIP (old string not found): {description}")
        RESULTS["skipped"] += 1
        return False

    backup(filepath)

    if count == 0:
        new_content = content.replace(old, new)
    else:
        new_content = content.replace(old, new, count)

    if write_file(filepath, new_content):
        action = "DRY-RUN" if DRY_RUN else "APPLIED"
        print(f"  {action}: {description}")
        RESULTS["applied"] += 1
        return True
    else:
        RESULTS["failed"] += 1
        return False


def patch_regex(filepath, pattern, replacement, description, count=0):
    """Apply a regex-based patch."""
    content = read_file(filepath)
    if content is None:
        RESULTS["failed"] += 1
        return False

    if not re.search(pattern, content):
        print(f"  SKIP (pattern not found): {description}")
        RESULTS["skipped"] += 1
        return False

    backup(filepath)

    if count > 0:
        new_content = re.sub(pattern, replacement, content, count=count)
    else:
        new_content = re.sub(pattern, replacement, content)

    if content == new_content:
        print(f"  SKIP (no change): {description}")
        RESULTS["skipped"] += 1
        return False

    if write_file(filepath, new_content):
        action = "DRY-RUN" if DRY_RUN else "APPLIED"
        print(f"  {action}: {description}")
        RESULTS["applied"] += 1
        return True
    else:
        RESULTS["failed"] += 1
        return False
