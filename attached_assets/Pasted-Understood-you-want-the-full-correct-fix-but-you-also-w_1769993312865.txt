B4) Hash function (server)

Use stable JSON stringify + sha256.

import crypto from "crypto";

export function stableStringify(obj: any): string {
  return JSON.stringify(obj, Object.keys(obj).sort());
}

export function sha256(text: string): string {
  return crypto.createHash("sha256").update(text).digest("hex");
}

(If nested ordering matters, use a stable stringify library; otherwise implement deep stable stringify.)

D) Simulation Engine Fixes (Correctness)

D1) Apply overrides deterministically

Implement function:

function applyOverrides(base: CompanyState["financials"], o: ScenarioOverrides) {
  const burn = base.monthlyBurn * (o.expenseMultiplier ?? 1);
  const growth = base.revenueGrowthRate + (o.revenueGrowthDelta ?? 0);
  const revenue = base.revenueMonthly * (1 + (o.pricingDelta ?? 0));
  return { burn, growth, revenue };
}

D2) Monte Carlo must generate true distributions

Pseudo:

function runMonteCarlo(input: SimulationInput): SimulationOutput {
  const { numPaths, horizonMonths } = input.config;
  const samplesRunway: number[] = [];
  const samplesCashEnd: number[] = [];
  const samplesBurn: number[] = [];

  const base = input.base;
  const adj = applyOverrides(base, input.overrides);

  for (let i=0; i<numPaths; i++) {
    // randomize uncertainty around growth/burn (simple gaussian or uniform)
    const burn = adj.burn * randBetween(0.95, 1.10);
    const growth = adj.growth + randBetween(-0.01, 0.01);
    let cash = base.cashBalance;
    let revenue = adj.revenue;

    let runway = 0;
    for (let m=1; m<=horizonMonths; m++) {
      revenue = revenue * (1 + growth);
      const net = revenue - burn;       // simplistic; adapt to your model
      cash += net;

      if (cash <= 0) { runway = m; break; }
      if (m === horizonMonths) runway = horizonMonths;
    }

    samplesRunway.push(runway);
    samplesCashEnd.push(cash);
    samplesBurn.push(burn);
  }

  const p = (arr:number[], q:number) => percentile(arr, q); // q=0.1/0.5/0.9
  const deterministic = variance(samplesRunway) === 0;

  const survival18 = samplesRunway.filter(x => x >= 18).length / samplesRunway.length;

  return {
    runId: input.runId,
    scenarioId: input.scenarioId,
    companyId: input.companyId,
    dataSnapshotId: input.dataSnapshotId,
    deterministic,
    metrics: {
      runwayMonths: { p10: p(samplesRunway, 0.1), p50: p(samplesRunway, 0.5), p90: p(samplesRunway, 0.9) },
      survival18mo: survival18,
      cashEnd: { p10: p(samplesCashEnd, 0.1), p50: p(samplesCashEnd, 0.5), p90: p(samplesCashEnd, 0.9) },
      burn: { p10: p(samplesBurn, 0.1), p50: p(samplesBurn, 0.5), p90: p(samplesBurn, 0.9) },
    },
    provenance: {
      updatedAt: input.asOf,
      source: "scenario",
      inputHash: sha256(stableStringify(input)),
    },
    debug: { numPaths, sampleCount: samplesRunway.length }
  };
}

Critical requirements:
